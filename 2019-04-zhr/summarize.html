► Solution:

flights %>%
  summarize(
    mean(arr_delay, na.rm = TRUE),
    mean(dep_delay, na.rm = TRUE),
    sd(arr_delay, na.rm = TRUE),
    sd(dep_delay, na.rm = TRUE)
  )
## # A tibble: 1 x 4

  ggplot(aes(_____)) +

► Solution: We need geom_col(), because the data is already aggregated and contains one row per geometry object we’re plotting.

mean_miles_by_carrier <-
  flights %>%
  group_by(carrier) %>% 
  summarize(mean_distance = mean(distance)) %>% 
  ungroup()

mean_miles_by_carrier %>%
  arrange(desc(mean_distance)) %>%
  mutate(carrier = fct_inorder(carrier)) %>%
  ggplot() +
  geom_col(aes(x = carrier, y = mean_distance))
