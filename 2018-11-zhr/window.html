Busiest month

monthly_shares <-
  flights %>%
  group_by(carrier, month) %>%
  summarize(distance = sum(distance)) %>%
  mutate(total_distance = sum(distance)) %>%
  ungroup() %>%
  mutate(month_share = distance / total_distance) %>%
  arrange(-total_distance)
